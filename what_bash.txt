__conda_setup="$('/home/adnan/anaconda3/bin/conda' 'shell.bash' 'hook' 2>/dev/null)"

    . "/home/adnan/anaconda3/etc/profile.d/conda.sh"
  else
    export PATH="/home/adnan/anaconda3/bin:$PATH"
